    ```js
    var str = ' a a bb c c dd'
    str = b.replace(/ /g,'')
    console.log(str) // 'aabbccdd'
    // 去除所有的空格
    str = str.replace(/\s*/g, "");
    // 去除两头空格
    str = str.replace(/^\s*|\s*$/g, "");
    // 去除左边的空格
    str = str.replace(/^\s*/, "");
    // 去除右边的空格
    str = str.replace(/(\s*$)/g, "");
    ```
